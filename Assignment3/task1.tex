\subsection*{Gini Index}

	
		\begin{equation}
		Gini(t) = 1 - \sum_{i=0}^{c-1} [p(i|t)]^{2} 
		\end{equation}
	
		c = number of classes \\
		p(i$|$t) = fraction of records belonging to class i
